/* for our LOCAL purposes we could use the most BASIC implementation of @font-face ever, 
	if we wanted to check out how a typeface we own in otf might look while we are sketching. 
	You would not do this if your site were published in any way. */

@font-face {
  font-family: "MyriadPro-Bold";
  src: url("type/MyriadPro-Bold.otf") format("opentype");
  }

@font-face {
  font-family: "MinionPro-Semibold";
  src: url("type/MinionPro-Semibold.otf") format("opentype");
  }

/* now the fonts have been brought in for use you can include them in the rules you write */
